package main

import "fmt"

func main() {
	var matriz [3][2]int
	for i := 0; i < 3; i++ {
		for j := 0; j < 2; j++ {
			fmt.Printf("digite o valor contido nas linhas: [%d] e informe o valor da coluna: [%d] .", i, j)
			fmt.Scanln(&matriz[i][j])
		}
	}
	for _, mat := range matriz {
		fmt.Print(mat)
	}
}
